# PROBLEM

Currently, all check items render as free-text inputs regardless of type. This is slow, error-prone, and doesn't match workshop workflows.

**Required fixes:**
1. RAG items â†’ Large Green/Amber/Red buttons
2. Tyre items â†’ Manufacturer, Size, Speed, Load + 3-point tread depth with sliders
3. Brake items â†’ Disc/Drum, Pad thickness, Disc thickness for N/S and O/S
4. Yes/No items â†’ Two large buttons
5. Measurement items â†’ Numeric input with unit
6. Select items â†’ Dropdown picker
7. Fluid level items â†’ Preset options (OK, Low, Very Low, Overfilled)

## 1.3 Component Specification

```typescript
// components/RAGSelector.tsx

interface RAGSelectorProps {
  value: 'green' | 'amber' | 'red' | null;
  onChange: (value: 'green' | 'amber' | 'red') => void;
  disabled?: boolean;
  size?: 'default' | 'large';  // large = 72px height
}

function RAGSelector({ value, onChange, size = 'large' }: RAGSelectorProps) {
  const buttons = [
    { value: 'green', label: 'PASS', icon: 'âœ“', color: 'bg-green-500' },
    { value: 'amber', label: 'ADVISORY', icon: 'âš ', color: 'bg-amber-500' },
    { value: 'red', label: 'URGENT', icon: 'âœ•', color: 'bg-red-500' },
  ];

  return (
    <div className="flex gap-3">
      {buttons.map((btn) => (
        <button
          key={btn.value}
          onClick={() => {
            onChange(btn.value);
            // Haptic feedback
            if (navigator.vibrate) navigator.vibrate(50);
          }}
          className={cn(
            'flex-1 flex flex-col items-center justify-center',
            'border-2 transition-all',
            size === 'large' ? 'h-[72px]' : 'h-[56px]',
            value === btn.value
              ? `${btn.color} text-white border-transparent`
              : 'bg-white text-gray-700 border-gray-300'
          )}
        >
          <span className="text-2xl">{btn.icon}</span>
          <span className="text-sm font-semibold">{btn.label}</span>
        </button>
      ))}
    </div>
  );
}
```

## 1.4 Behavior
- Tap to select (single selection)
- Selected button fills with color
- Haptic feedback on selection
- Notes field appears below (optional)
- Photo button always visible

## 3.3 Numeric Picker Component

```typescript
// components/NumericPicker.tsx

interface NumericPickerProps {
  value: number;
  onChange: (value: number) => void;
  min: number;
  max: number;
  step?: number;
  unit?: string;
  thresholds?: {
    red: number;
    amber: number;
  };
}

function NumericPicker({ value, onChange, min, max, step = 1, unit, thresholds }: NumericPickerProps) {
  const options = [];
  for (let i = min; i <= max; i += step) {
    options.push(i);
  }

  const getColor = () => {
    if (!thresholds) return 'default';
    if (value < thresholds.red) return 'red';
    if (value < thresholds.amber) return 'amber';
    return 'green';
  };

  return (
    <div className="relative">
      <select
        value={value}
        onChange={(e) => onChange(parseFloat(e.target.value))}
        className={cn(
          'h-12 w-20 text-center text-lg font-mono border-2 appearance-none',
          getColor() === 'red' && 'bg-red-100 border-red-500',
          getColor() === 'amber' && 'bg-amber-100 border-amber-500',
          getColor() === 'green' && 'bg-green-100 border-green-500',
          getColor() === 'default' && 'bg-white border-gray-300',
        )}
      >
        {options.map((opt) => (
          <option key={opt} value={opt}>{opt}</option>
        ))}
      </select>
      {unit && <span className="ml-2 text-gray-600">{unit}</span>}
    </div>
  );
}
```

# 5. ADMIN SETTINGS â€” TYRE DATA

## 5.1 Database Schema

```sql
-- Tyre Manufacturers (admin managed)
CREATE TABLE tyre_manufacturers (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  organization_id UUID NOT NULL REFERENCES organizations(id) ON DELETE CASCADE,
  name VARCHAR(100) NOT NULL,
  is_active BOOLEAN DEFAULT true,
  sort_order INTEGER DEFAULT 0,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_tyre_manufacturers_org ON tyre_manufacturers(organization_id);

-- Tyre Sizes (admin managed)
CREATE TABLE tyre_sizes (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  organization_id UUID NOT NULL REFERENCES organizations(id) ON DELETE CASCADE,
  size VARCHAR(20) NOT NULL,  -- e.g., "225/45R17"
  width INTEGER,              -- e.g., 225
  profile INTEGER,            -- e.g., 45
  rim_size INTEGER,           -- e.g., 17
  is_active BOOLEAN DEFAULT true,
  sort_order INTEGER DEFAULT 0,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_tyre_sizes_org ON tyre_sizes(organization_id);

-- Speed Ratings (pre-seeded, org-wide)
CREATE TABLE speed_ratings (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  code VARCHAR(5) NOT NULL,   -- e.g., "W"
  max_speed_kmh INTEGER,      -- e.g., 270
  max_speed_mph INTEGER,      -- e.g., 168
  description VARCHAR(50),
  sort_order INTEGER DEFAULT 0
);

-- Load Ratings (pre-seeded, org-wide)
CREATE TABLE load_ratings (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  code VARCHAR(5) NOT NULL,   -- e.g., "94"
  max_load_kg INTEGER,        -- e.g., 670
  sort_order INTEGER DEFAULT 0
);
```

# 8. API ENDPOINTS

## 8.1 Tyre Manufacturers

```
GET    /api/v1/tyre-manufacturers         # List all (filtered by org)
POST   /api/v1/tyre-manufacturers         # Create new
PATCH  /api/v1/tyre-manufacturers/:id     # Update
DELETE /api/v1/tyre-manufacturers/:id     # Delete
POST   /api/v1/tyre-manufacturers/reorder # Reorder
POST   /api/v1/tyre-manufacturers/seed    # Seed defaults
```

## 8.2 Tyre Sizes

```
GET    /api/v1/tyre-sizes                 # List all (with filters)
POST   /api/v1/tyre-sizes                 # Create new
PATCH  /api/v1/tyre-sizes/:id             # Update
DELETE /api/v1/tyre-sizes/:id             # Delete
POST   /api/v1/tyre-sizes/seed            # Seed common sizes
```

## 8.3 Reference Data (Read-only)

```
GET    /api/v1/speed-ratings              # List speed ratings
GET    /api/v1/load-ratings               # List load ratings
```

## 8.4 Thresholds

```
GET    /api/v1/sites/:id/thresholds       # Get site thresholds
PATCH  /api/v1/sites/:id/thresholds       # Update thresholds
POST   /api/v1/sites/:id/thresholds/reset # Reset to defaults
```
